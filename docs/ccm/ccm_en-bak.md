## 
- 
-  <i>sudo ln -s /full/path/to/your/file/ccm.sh /usr/local/bin/ccm</i>


<p id="ccm-help">&nbsp;</p>

## 
-  <i>ccm help</i>


<p id="ccm-sample">&nbsp;</p>

## 
-  <i>ccm install flora</i>
-  <i>ccm start flora</i>
-  <i>ccm stop flora</i>
-  <i>ccm restart flora</i>
-  <i>ccm upgrade flora</i>
-  <i>ccm uninstall flora</i>


<p id="ccm-all">&nbsp;</p>
